### **2. Add Conversation Memory** ✅ MAKES IT FEEL "SMART"

```csharp
// Add this method to ChatUseCase
private async Task<string> BuildPromptWithMemory(
    string question, 
    string userId,
    CancellationToken ct)
{
    // Get last 5 conversations
    var recentChats = await _chatRepo.GetRecentAsync(userId, 5, ct);
    
    if (!recentChats.Any())
    {
        return question; // First conversation
    }
    
    // Build memory context
    var memoryContext = string.Join("\n", recentChats.Select(c => 
        $"User: {c.Question}\nAssistant: {c.Answer}"));
    
    return $@"Previous conversation:
{memoryContext}

Current question: {question}";
}

// Use in ChatAsync:
var questionWithMemory = await BuildPromptWithMemory(question, userId, cancellationToken);
var queryVector = await _llmService.EmbeddingAsync(questionWithMemory, cancellationToken);
```

**Expected Result**: System "remembers" context across conversations!

// Modify prompt to include citation markers
private string BuildRagPromptWithCitations(string context, string question)
{
    var chunks = context.Split("\n\n");
    var numberedContext = string.Join("\n\n", 
        chunks.Select((c, i) => $"[{i+1}] {c}"));
    
    return $@"Context (with source numbers):
{numberedContext}

Question: {question}

Answer the question using the context above. 
Cite sources using [1], [2] format after each claim.
Only use information from the provided context.

Answer:";
}

## **Full V1.0 ChatUseCase (with all improvements)**

```csharp
public async Task<(string Answer, List<Citation> Citations, decimal CostSeconds)> ChatV1Async(
    string userId,
    string question,
    CancellationToken cancellationToken = default)
{
    var stopwatch = System.Diagnostics.Stopwatch.StartNew();
    
    try
    {
        // 1. Load conversation memory
        var conversationMemory = await LoadConversationMemory(userId, cancellationToken);
        _logger.LogInformation("Loaded {Count} previous messages", conversationMemory.Count);
        
        // 2. Query rewriting (HyDE)
        var hypotheticalDoc = await _llmService.ChatAsync(
            $"Write a detailed answer to: {question}", cancellationToken);
        
        // 3. Multi-query expansion
        var similarQuestions = await GenerateSimilarQuestions(question, cancellationToken);
        var allQueryVectors = new List<float[]>();
        
        foreach (var q in similarQuestions.Take(3))
        {
            var vec = await _llmService.EmbeddingAsync(q, cancellationToken);
            allQueryVectors.Add(vec);
        }
        
        // 4. Hybrid retrieval with fusion
        var allResults = new List<List<DocumentChunk>>();
        foreach (var vec in allQueryVectors)
        {
            var results = await _vectorStore.SearchAsync(
                collectionName, vec, 20, filter, cancellationToken);
            allResults.Add(results);
        }
        
        var fusedChunks = FuseMultipleResults(allResults);
        var validChunks = FilterValidChunks(fusedChunks);
        
        // 5. Contextual expansion (spreading activation)
        var expandedChunks = await ExpandContext(validChunks, cancellationToken);
        
        // 6. Reranking
        var finalChunks = await _rerankService.RerankAsync(
            question, expandedChunks, take: 5, cancellationToken);
        
        // 7. Build prompt with citations
        var prompt = BuildPromptWithCitations(finalChunks, question, conversationMemory);
        
        // 8. Generate answer
        var answer = await _llmService.ChatAsync(prompt, cancellationToken);
        
        // 9. Self-reflection & correction
        var validatedAnswer = await SelfCorrect(question, answer, finalChunks, cancellationToken);
        
        // 10. Extract citations
        var citations = ExtractCitations(validatedAnswer, finalChunks);
        
        stopwatch.Stop();
        
        // 11. Update memory
        await UpdateConversationMemory(userId, question, validatedAnswer, cancellationToken);
        
        _logger.LogInformation("✅ V1.0 RAG complete in {Time:F2}s", stopwatch.Elapsed.TotalSeconds);
        
        return (validatedAnswer, citations, (decimal)stopwatch.Elapsed.TotalSeconds);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, "V1.0 RAG failed");
        throw;
    }
}
```

## **Database Migration for V1.0**

```sql
-- Add conversation memory table
CREATE TABLE user_conversation_memory (
    Id CHAR(36) PRIMARY KEY,
    UserId VARCHAR(100) NOT NULL,
    Question TEXT NOT NULL,
    Answer TEXT NOT NULL,
    Timestamp DATETIME NOT NULL,
    SessionId CHAR(36),
    INDEX IX_UserId_Timestamp (UserId, Timestamp)
);

-- Add episodic memory (vector-indexed past interactions)
CREATE TABLE user_episodic_memory (
    Id CHAR(36) PRIMARY KEY,
    UserId VARCHAR(100) NOT NULL,
    Content TEXT NOT NULL,
    Vector BLOB, -- Store embedding
    Metadata JSON,
    Timestamp DATETIME NOT NULL,
    INDEX IX_UserId (UserId)
);

-- Add knowledge graph nodes
CREATE TABLE knowledge_graph_nodes (
    Id CHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL,
    Type VARCHAR(50), -- person, place, concept, etc.
    Properties JSON,
    INDEX IX_Name (Name),
    INDEX IX_Type (Type)
);

-- Add knowledge graph edges
CREATE TABLE knowledge_graph_edges (
    Id CHAR(36) PRIMARY KEY,
    SubjectId CHAR(36) NOT NULL,
    Predicate VARCHAR(100) NOT NULL,
    ObjectId CHAR(36) NOT NULL,
    Weight FLOAT DEFAULT 1.0,
    FOREIGN KEY (SubjectId) REFERENCES knowledge_graph_nodes(Id),
    FOREIGN KEY (ObjectId) REFERENCES knowledge_graph_nodes(Id),
    INDEX IX_Subject (SubjectId),
    INDEX IX_Object (ObjectId)
);
```
